/*
 * 1. 找一个基准点
 * 2. 建立两个数组，分别存储左边和右边的数组
 * 3. 利递归进行比较
 */

function quickSort(arr){

    if(arr.length <= 1){
        return arr;
    }

    var num = Math.floor(arr.length / 2);

    var numValue = arr.splice(num, 1);

    var left = [];
    var right = [];

    for(var i=0,len=arr.length; i<len; i++) {
        if(arr[i] < numValue){
            left.push(arr[i]);
        }else{
            right.push(arr[i]);
        }
    }
    return quickSort(left).concat([numValue], quickSort(right));
}

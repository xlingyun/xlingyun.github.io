### 深拷贝
/*
 * 递归：
 * 1. 函数调用自身，执行递的动作
 * 2. 最后一次判断一个终止条件
 */

function deepCopy(obj){
    
    var newObj = {};

    if(typeof obj != 'object'){
        <!-- console.trace();  -->//打印执行过程
        return obj;
    }
    
    for(var attr in obj){
        newObj[attr] = deepCopy(obj[attr])
    }

    return newObj;
}

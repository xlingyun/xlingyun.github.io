function createJs(sUrl) {
    var oScript = document.createElement('script');
    oScript.type = 'text/javascript';
    oScript.src = sUrl;
    document.getElementsByTagName('head')[0].appendChild(oScript);
}

// createJs('jsonp.js');
createJs('jsonp.js?callback=fn')
